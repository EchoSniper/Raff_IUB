// Pin assignments for LM35 sensors
const int lm35Pin1 = A0; // First LM35 connected to analog pin A0
const int lm35Pin2 = A1; // Second LM35 connected to analog pin A1

void setup() {
  Serial.begin(9600); // Initialize the Serial communication at 9600 baud
  Serial.println("Timestamp,Temperature1,Temperature2"); // CSV headers for data logging
}

void loop() {
  // Read raw analog values from the sensors
  int rawValue1 = analogRead(lm35Pin1);
  int rawValue2 = analogRead(lm35Pin2);
  
  // Convert analog readings to temperature in Celsius
  float temperature1 = (rawValue1 * 5.0 / 1024.0) * 100.0; // LM35 gives 10mV per degree Celsius
  float temperature2 = (rawValue2 * 5.0 / 1024.0) * 100.0;

  // Get current timestamp in milliseconds
  unsigned long timestamp = millis();

  // Send data in CSV format: Timestamp,Temperature1,Temperature2
  Serial.print(timestamp);
  Serial.print(",");
  Serial.print(temperature1);
  Serial.print(",");
  Serial.println(temperature2);

  delay(1000); // Wait for 1 second before the next reading
}
